CREATE OR REPLACE FUNCTION notify_OLD()
RETURNS TRIGGER AS
$BODY$
DECLARE
BEGIN
  PERFORM pg_notify(TG_TABLE_NAME,
    TG_WHEN || ' ' || TG_OP || ' ' || TG_TABLE_NAME
    || ' Old Row:' || row_to_json(OLD));
  RETURN OLD;
END;
$BODY$ LANGUAGE plpgsql;

CREATE OR REPLACE FUNCTION notify_NEW()
RETURNS TRIGGER AS
$BODY$
DECLARE
BEGIN
  PERFORM pg_notify(TG_TABLE_NAME,
    TG_WHEN || ' ' || TG_OP || ' ' || TG_TABLE_NAME
    || ' New Row:' || row_to_json(NEW));
  RETURN NEW;
END;
$BODY$ LANGUAGE plpgsql;
